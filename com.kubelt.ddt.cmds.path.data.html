  {:command "data"
   :desc "Print the data path"
   :requiresArg false
   :builder (fn [^Yargs yargs]
              yargs)
   :handler (fn [args]
              (let [{app-name :\$0}
                    (js->clj args :keywordize-keys true)]
                (println (path/data app-name))))})
